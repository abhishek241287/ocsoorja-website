import { ENGINEERING_CONSTANTS, getPeakSunHours, roundToStandardSize } from "../constants";

export interface SolarInputs {
  monthlyBill: number;
  tariff: number;
  state: string;
  roofArea?: number;
  performanceRatio?: number;
  panelDegradation?: number;
  temperatureLoss?: number;
  dustLoss?: number;
  shadingLoss?: number;
  orientationLoss?: number;
}

export interface SolarStep {
  label: string;
  value: string;
  unit?: string;
}

export interface SolarResult {
  recommendedCapacity: number;
  standardSize: number;
  dailyGeneration: number;
  monthlyGeneration: number;
  monthlySavings: number;
  areaRequired: number;
  steps: SolarStep[];
}

export function calculateSolarSizing(inputs: SolarInputs): SolarResult {
  const {
    monthlyBill,
    tariff,
    state,
    performanceRatio = ENGINEERING_CONSTANTS.solar.defaultPerformanceRatio,
    temperatureLoss = ENGINEERING_CONSTANTS.solar.defaultTemperatureLoss,
    dustLoss = ENGINEERING_CONSTANTS.solar.defaultDustLoss,
    shadingLoss = 0,
    orientationLoss = 0,
  } = inputs;

  const peakSunHours = getPeakSunHours(state);
  
  // Step 1: Monthly consumption
  const monthlyConsumption = monthlyBill / tariff;
  
  // Step 2: Daily consumption
  const dailyConsumption = monthlyConsumption / 30;
  
  // Step 3: Adjusted performance ratio
  const adjustedPR = performanceRatio * (1 - temperatureLoss) * (1 - dustLoss) * (1 - shadingLoss) * (1 - orientationLoss);
  
  // Step 4: Raw capacity needed
  const rawCapacity = dailyConsumption / (peakSunHours * adjustedPR);
  
  // Step 5: Standard size
  const standardSize = roundToStandardSize(rawCapacity, ENGINEERING_CONSTANTS.solar.standardSizes);
  
  // Results
  const dailyGeneration = standardSize * peakSunHours * adjustedPR;
  const monthlyGeneration = dailyGeneration * 30;
  const monthlySavings = monthlyGeneration * tariff;
  const areaRequired = standardSize * ENGINEERING_CONSTANTS.solar.areaPerKw;

  const steps: SolarStep[] = [
    { label: "Monthly Bill", value: `₹ ${monthlyBill.toLocaleString()}`, unit: "INR" },
    { label: "Electricity Tariff", value: `₹ ${tariff}`, unit: "per kWh" },
    { label: "Monthly Consumption", value: monthlyConsumption.toFixed(1), unit: "kWh" },
    { label: "Daily Consumption", value: dailyConsumption.toFixed(1), unit: "kWh" },
    { label: "Peak Sun Hours", value: peakSunHours.toFixed(1), unit: "hours/day" },
    { label: "Performance Ratio", value: adjustedPR.toFixed(2), unit: "" },
    { label: "Raw Capacity Needed", value: rawCapacity.toFixed(2), unit: "kW" },
    { label: "Nearest Standard Size", value: standardSize.toString(), unit: "kW" },
  ];

  return {
    recommendedCapacity: rawCapacity,
    standardSize,
    dailyGeneration,
    monthlyGeneration,
    monthlySavings,
    areaRequired,
    steps,
  };
}

import { ENGINEERING_CONSTANTS, getBatteryChemistry, roundToStandardSize } from "../constants";
import type { BatteryChemistry } from "../constants";

export interface ApplianceLoad {
  name: string;
  wattage: number;
  quantity: number;
}

export interface BatteryInputs {
  appliances: ApplianceLoad[];
  backupHours: number;
  voltage: number;
  chemistry: BatteryChemistry;
  reserveMargin?: number;
}

export interface BatteryStep {
  label: string;
  value: string;
  unit?: string;
}

export interface BatteryResult {
  totalLoad: number;
  totalEnergy: number;
  requiredCapacityWh: number;
  requiredCapacityAh: number;
  standardCapacityKwh: number;
  standardCapacityAh: number;
  actualBackupTime: number;
  steps: BatteryStep[];
}

export function calculateBatteryBackup(inputs: BatteryInputs): BatteryResult {
  const { appliances, backupHours, voltage, chemistry, reserveMargin = 0 } = inputs;
  
  const batterySpecs = getBatteryChemistry(chemistry);
  const dod = batterySpecs.dod;
  const efficiency = batterySpecs.efficiency;

  // Step 1: Total load
  const totalLoad = appliances.reduce((sum, app) => sum + (app.wattage * app.quantity), 0);
  
  // Step 2: Total energy needed
  const totalEnergy = totalLoad * backupHours;
  
  // Step 3: Account for DoD and efficiency
  const requiredCapacityWh = totalEnergy / (dod * efficiency);
  const requiredCapacityAh = requiredCapacityWh / voltage;
  
  // Step 4: Add reserve margin
  const capacityWithReserve = requiredCapacityWh * (1 + reserveMargin);
  
  // Step 5: Standard size
  const standardCapacityKwh = roundToStandardSize(capacityWithReserve / 1000, ENGINEERING_CONSTANTS.battery.standardCapacitiesKwh);
  const standardCapacityAh = roundToStandardSize(requiredCapacityAh, ENGINEERING_CONSTANTS.battery.standardCapacitiesAh);
  
  // Actual backup time
  const actualBackupTime = (standardCapacityKwh * 1000 * dod * efficiency) / totalLoad;

  const steps: BatteryStep[] = [
    { label: "Total Load", value: totalLoad.toString(), unit: "W" },
    { label: "Backup Duration", value: backupHours.toString(), unit: "hours" },
    { label: "Total Energy Needed", value: totalEnergy.toString(), unit: "Wh" },
    { label: "Depth of Discharge", value: `${(dod * 100).toFixed(0)}%`, unit: "" },
    { label: "Battery Efficiency", value: `${(efficiency * 100).toFixed(0)}%`, unit: "" },
    { label: "Required Capacity", value: requiredCapacityWh.toFixed(0), unit: "Wh" },
    { label: "Required Capacity", value: requiredCapacityAh.toFixed(0), unit: "Ah" },
    { label: "Standard Capacity", value: standardCapacityKwh.toString(), unit: "kWh" },
  ];

  return {
    totalLoad,
    totalEnergy,
    requiredCapacityWh,
    requiredCapacityAh,
    standardCapacityKwh,
    standardCapacityAh,
    actualBackupTime,
    steps,
  };

import { ENGINEERING_CONSTANTS, roundToStandardSize } from "../constants";

export interface InverterAppliance {
  name: string;
  wattage: number;
  quantity: number;
  isMotor?: boolean;
}

export interface InverterInputs {
  appliances: InverterAppliance[];
  futureExpansion: "none" | "low" | "high";
  powerFactor?: number;
  surgeFactor?: number;
  diversityFactor?: number;
}

export interface InverterStep {
  label: string;
  value: string;
  unit?: string;
}

export interface InverterResult {
  runningLoad: number;
  peakSurgeLoad: number;
  requiredCapacity: number;
  standardSize: number;
  futureHeadroom: number;
  steps: InverterStep[];
}

export function calculateInverterSizing(inputs: InverterInputs): InverterResult {
  const {
    appliances,
    futureExpansion,
    powerFactor = ENGINEERING_CONSTANTS.inverter.powerFactor,
    surgeFactor = ENGINEERING_CONSTANTS.inverter.surgeFactor,
    diversityFactor = 1.0,
  } = inputs;

  const futureFactor = ENGINEERING_CONSTANTS.inverter.futureExpansion[futureExpansion];

  // Step 1: Running load
  const runningLoad = appliances.reduce((sum, app) => sum + (app.wattage * app.quantity), 0);
  
  // Step 2: Surge load (motor starting)
  const motorLoad = appliances
    .filter((app) => app.isMotor)
    .reduce((sum, app) => sum + (app.wattage * app.quantity * surgeFactor), 0);
  const nonMotorLoad = appliances
    .filter((app) => !app.isMotor)
    .reduce((sum, app) => sum + (app.wattage * app.quantity), 0);
  const peakSurgeLoad = motorLoad + nonMotorLoad;
  
  // Step 3: Apply diversity and future expansion
  const adjustedLoad = Math.max(runningLoad, peakSurgeLoad) * diversityFactor * futureFactor;
  
  // Step 4: Convert to kVA
  const requiredCapacity = adjustedLoad / powerFactor / 1000;
  
  // Step 5: Standard size
  const standardSize = roundToStandardSize(requiredCapacity, ENGINEERING_CONSTANTS.inverter.standardSizes);
  
  const futureHeadroom = ((futureFactor - 1) * 100).toFixed(0);

  const steps: InverterStep[] = [
    { label: "Running Load", value: runningLoad.toString(), unit: "W" },
    { label: "Peak Surge Load", value: peakSurgeLoad.toString(), unit: "W" },
    { label: "Diversity Factor", value: diversityFactor.toFixed(2), unit: "" },
    { label: "Future Expansion", value: `${futureHeadroom}%`, unit: "" },
    { label: "Required Capacity", value: requiredCapacity.toFixed(2), unit: "kVA" },
    { label: "Standard Size", value: standardSize.toString(), unit: "kVA" },
  ];

  return {
    runningLoad,
    peakSurgeLoad,
    requiredCapacity,
    standardSize,
    futureHeadroom: parseInt(futureHeadroom),
    steps,
  };
}

import { ENGINEERING_CONSTANTS, roundToStandardSize } from "../constants";

export interface CableInputs {
  current: number;
  distance: number;
  voltage: number;
  maxVoltageDrop: number;
  cableType: "copper" | "aluminum";
  installation: "conduit" | "tray" | "underground" | "aerial";
  ambientTemperature?: number;
}

export interface CableStep {
  label: string;
  value: string;
  unit?: string;
}

export interface CableResult {
  requiredSize: number;
  standardSize: number;
  actualVoltageDrop: number;
  actualVoltageDropPercent: number;
  cableAmpacity: number;
  safetyMargin: number;
  steps: CableStep[];
}

export function calculateCableSizing(inputs: CableInputs): CableResult {
  const {
    current,
    distance,
    voltage,
    maxVoltageDrop,
    cableType,
    installation,
  } = inputs;

  const resistivity = cableType === "copper" 
    ? ENGINEERING_CONSTANTS.cable.copperResistivity 
    : ENGINEERING_CONSTANTS.cable.aluminumResistivity;
  
  const derating = ENGINEERING_CONSTANTS.cable.ampacityDerating[installation];

  // Step 1: Calculate required cross-section for voltage drop
  const maxDropVolts = voltage * (maxVoltageDrop / 100);
  const requiredSize = (2 * current * distance * resistivity) / maxDropVolts;
  
  // Step 2: Round to standard size
  const standardSize = roundToStandardSize(requiredSize, ENGINEERING_CONSTANTS.cable.standardSizes);
  
  // Step 3: Calculate actual voltage drop with standard size
  const actualVoltageDrop = (2 * current * distance * resistivity) / standardSize;
  const actualVoltageDropPercent = (actualVoltageDrop / voltage) * 100;
  
  // Step 4: Estimate ampacity (simplified)
  const baseAmpacity = standardSize * 8; // rough estimate: 8A per sq mm
  const cableAmpacity = baseAmpacity * derating;
  
  // Step 5: Safety margin
  const safetyMargin = ((cableAmpacity - current) / current) * 100;

  const steps: CableStep[] = [
    { label: "Current", value: current.toString(), unit: "A" },
    { label: "Distance", value: distance.toString(), unit: "m" },
    { label: "Voltage", value: voltage.toString(), unit: "V" },
    { label: "Max Voltage Drop", value: `${maxVoltageDrop}%`, unit: "" },
    { label: "Resistivity", value: resistivity.toString(), unit: "Ω·mm²/m" },
    { label: "Required Size", value: requiredSize.toFixed(2), unit: "sq mm" },
    { label: "Standard Size", value: standardSize.toString(), unit: "sq mm" },
    { label: "Actual Voltage Drop", value: actualVoltageDropPercent.toFixed(2), unit: "%" },
  ];

  return {
    requiredSize,
    standardSize,
    actualVoltageDrop,
    actualVoltageDropPercent,
    cableAmpacity,
    safetyMargin,
    steps,
  };
}

import { ENGINEERING_CONSTANTS } from "../constants";

export interface ROIInputs {
  systemCost: number;
  systemCapacity: number;
  electricityTariff: number;
  annualGeneration?: number;
  tariffEscalation?: number;
  maintenanceCost?: number;
  analysisPeriod?: number;
}

export interface YearlyData {
  year: number;
  generation: number;
  savings: number;
  cumulativeSavings: number;
  maintenance: number;
  netSavings: number;
}

export interface ROIStep {
  label: string;
  value: string;
  unit?: string;
}

export interface ROIResult {
  paybackPeriod: number;
  tenYearSavings: number;
  totalSavings: number;
  roi: number;
  npv: number;
  yearlyData: YearlyData[];
  steps: ROIStep[];
}

export function calculateROI(inputs: ROIInputs): ROIResult {
  const {
    systemCost,
    systemCapacity,
    electricityTariff,
    annualGeneration,
    tariffEscalation = ENGINEERING_CONSTANTS.financial.defaultTariffEscalation,
    maintenanceCost = ENGINEERING_CONSTANTS.financial.defaultMaintenanceCost,
    analysisPeriod = ENGINEERING_CONSTANTS.financial.defaultAnalysisPeriod,
  } = inputs;

  const peakSunHours = 5.0; // default
  const performanceRatio = ENGINEERING_CONSTANTS.solar.defaultPerformanceRatio;
  
  // Calculate generation if not provided
  const actualAnnualGeneration = annualGeneration || (systemCapacity * peakSunHours * 365 * performanceRatio);
  
  const yearlyMaintenance = systemCost * maintenanceCost;
  
  const yearlyData: YearlyData[] = [];
  let cumulativeSavings = 0;
  let paybackPeriod = 0;
  let paybackFound = false;

  for (let year = 1; year <= analysisPeriod; year++) {
    const tariff = electricityTariff * Math.pow(1 + tariffEscalation, year - 1);
    const savings = actualAnnualGeneration * tariff;
    const netSavings = savings - yearlyMaintenance;
    cumulativeSavings += netSavings;
    
    if (!paybackFound && cumulativeSavings >= systemCost) {
      paybackPeriod = year - 1 + (systemCost - (cumulativeSavings - netSavings)) / netSavings;
      paybackFound = true;
    }
    
    yearlyData.push({
      year,
      generation: actualAnnualGeneration,
      savings,
      cumulativeSavings,
      maintenance: yearlyMaintenance,
      netSavings,
    });
  }

  const totalSavings = cumulativeSavings;
  const roi = ((totalSavings - systemCost) / systemCost) * 100;
  const tenYearSavings = yearlyData.slice(0, 10).reduce((sum, y) => sum + y.netSavings, 0);

  const steps: ROIStep[] = [
    { label: "System Cost", value: `₹ ${systemCost.toLocaleString()}`, unit: "INR" },
    { label: "System Capacity", value: systemCapacity.toString(), unit: "kW" },
    { label: "Annual Generation", value: actualAnnualGeneration.toFixed(0), unit: "kWh" },
    { label: "Tariff Escalation", value: `${(tariffEscalation * 100).toFixed(0)}%`, unit: "per year" },
    { label: "Payback Period", value: paybackPeriod.toFixed(1), unit: "years" },
    { label: "10-Year Savings", value: `₹ ${tenYearSavings.toLocaleString()}`, unit: "INR" },
    { label: "Total ROI", value: `${roi.toFixed(0)}%`, unit: "" },
  ];

  return {
    paybackPeriod,
    tenYearSavings,
    totalSavings,
    roi,
    npv: 0, // simplified
    yearlyData,
    steps,
  };
}

  monthlyCost: number;
  rangePerHour: number;
  fullChargeTime: number;
  standardChargerSize: number;

export function calculateEVCharging(inputs: EVInputs): EVResult {

    batterySize,
    dailyCommute,
    efficiency = ENGINEERING_CONSTANTS.ev.defaultEfficiency,
    chargerType,
    electricityTariff,
  } = inputs;

  const chargerPowerMap: Record<string, number> = {
    "ac-3.3": 3.3,
    "ac-7.2": 7.2,
    "ac-11": 11,
    "ac-22": 22,
    "dc-50": 50,
    "dc-120": 120,
  };

  const chargerPower = chargerPowerMap[chargerType];
  
  // Step 1: Daily energy needed
  const dailyEnergyNeeded = dailyCommute / efficiency;

  // Step 6: Standard size

// src/pages/engineering-tools/solar-sizing.tsx
// Example calculator page — Solar Sizing

import { useState } from "react";
import { CalculatorLayout } from "@/components/engineering/CalculatorLayout";
import { CalculationSteps } from "@/components/engineering/CalculationSteps";
import { ResultCard } from "@/components/engineering/ResultCard";
import { calculateSolarSizing, type SolarInputs, type SolarResult } from "@/lib/engineering/calculations/solar-sizing";
import { ENGINEERING_CONSTANTS } from "@/lib/engineering/constants";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";

const STATES = Object.keys(ENGINEERING_CONSTANTS.solar.peakSunHours);

export default function SolarSizingCalculator() {
  const [result, setResult] = useState<SolarResult | null>(null);
  const [inputs, setInputs] = useState<Partial<SolarInputs>>({});

  const handleCalculate = (mode: "basic" | "advanced") => {
    const calculationInputs: SolarInputs = {
      monthlyBill: Number(inputs.monthlyBill) || 0,
      tariff: Number(inputs.tariff) || 7,
      state: inputs.state || "Uttar Pradesh",
      performanceRatio: mode === "advanced" ? Number(inputs.performanceRatio) : undefined,
      temperatureLoss: mode === "advanced" ? Number(inputs.temperatureLoss) : undefined,
      dustLoss: mode === "advanced" ? Number(inputs.dustLoss) : undefined,
    };

    const calcResult = calculateSolarSizing(calculationInputs);
    setResult(calcResult);
  };

  return (
    <CalculatorLayout
      title="Solar Sizing Calculator"
      description="Calculate the right solar capacity for your electricity needs."
      onCalculate={handleCalculate}
      result={result && (
        <div className="space-y-6">
          <ResultCard
            title="Results"
            results={[
              { label: "Recommended Solar Capacity", value: result.standardSize.toString(), unit: "kW", highlight: true },
              { label: "Estimated Daily Generation", value: result.dailyGeneration.toFixed(1), unit: "kWh" },
              { label: "Estimated Monthly Generation", value: result.monthlyGeneration.toFixed(0), unit: "kWh" },
              { label: "Estimated Monthly Savings", value: `₹ ${result.monthlySavings.toFixed(0)}`, unit: "INR" },
              { label: "Roof Area Required", value: result.areaRequired.toFixed(0), unit: "sq ft" },
            ]}
          />
          <CalculationSteps steps={result.steps} />
        </div>
      )}
    >
      {(mode) => (
        <div className="space-y-4">
          <div>
            <Label>Monthly Electricity Bill (₹) *</Label>
            <Input
              type="number"
              placeholder="8000"
              value={inputs.monthlyBill || ""}
              onChange={(e) => setInputs({ ...inputs, monthlyBill: e.target.value })}
            />
          </div>

          <div>
            <Label>State *</Label>
            <select
              className="w-full rounded-md border border-input px-3 py-2"
              value={inputs.state || ""}
              onChange={(e) => setInputs({ ...inputs, state: e.target.value })}
            >
              <option value="">Select state...</option>
              {STATES.map((state) => (
                <option key={state} value={state}>{state}</option>
              ))}
            </select>
          </div>

          <div>
            <Label>Average Electricity Tariff (₹/kWh)</Label>
            <Input
              type="number"
              step="0.1"
              placeholder="7.0"
              value={inputs.tariff || ""}
              onChange={(e) => setInputs({ ...inputs, tariff: e.target.value })}
            />
          </div>

          {mode === "advanced" && (
            <>
              <div>
                <Label>Performance Ratio</Label>
                <Input
                  type="number"
                  step="0.01"
                  placeholder="0.80"
                  value={inputs.performanceRatio || ""}
                  onChange={(e) => setInputs({ ...inputs, performanceRatio: e.target.value })}
                />
              </div>
              <div>
                <Label>Temperature Loss (%)</Label>
                <Input
                  type="number"
                  placeholder="10"
                  value={inputs.temperatureLoss || ""}
                  onChange={(e) => setInputs({ ...inputs, temperatureLoss: e.target.value })}
                />
              </div>
              <div>
                <Label>Dust/Soiling Loss (%)</Label>
                <Input
                  type="number"
                  placeholder="5"
                  value={inputs.dustLoss || ""}
                  onChange={(e) => setInputs({ ...inputs, dustLoss: e.target.value })}
                />
              </div>
            </>
          )}
        </div>
      )}
    </CalculatorLayout>
  );
}
