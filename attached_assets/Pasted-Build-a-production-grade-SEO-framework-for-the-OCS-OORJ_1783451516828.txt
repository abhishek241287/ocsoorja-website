import { brand, BRAND_OPENING_HOURS_SPEC } from "@/data/brand";

export function generateCanonical(path: string): string {
  const cleanPath = path.startsWith("/") ? path : `/${path}`;
  return `${SITE.url}${cleanPath}`;
}

export interface PageMeta { title: string; description: string; canonical: string; ogImage?: string; ogType?: string; twitterCard?: string; robots?: string; }

export function buildPageMeta(pageKey: keyof typeof PAGE_META, overrides: Partial<PageMeta> & { param?: string; param2?: string } = {}, currentPath: string = "/"): PageMeta {
  const preset = PAGE_META[pageKey];
  const canonical = generateCanonical(currentPath);
  let title = "title" in preset ? preset.title : "";
  let description = preset.description;
  if ("titleTemplate" in preset && overrides.param) title = preset.titleTemplate.replace("%s", overrides.param);
  if ("descriptionTemplate" in preset && overrides.param) {
    description = preset.descriptionTemplate.replace("%s", overrides.param);
    if (overrides.param2) description = description.replace("%s", overrides.param2);
  }
  return { title: overrides.title ?? title, description: overrides.description ?? description, canonical: overrides.canonical ?? canonical, ogImage: overrides.ogImage ?? SITE.ogImage, ogType: overrides.ogType ?? ("ogType" in preset ? preset.ogType : SEO_DEFAULTS.ogType), twitterCard: overrides.twitterCard ?? SEO_DEFAULTS.twitterCard, robots: overrides.robots ?? SEO_DEFAULTS.robots };
}

export function generateProductSchema(product: Product) {
  const schema: Record<string, unknown> = { "@context": SCHEMA_CONTEXT, "@type": "Product", "@id": `${SITE.url}/products/${product.slug}#product`, name: product.name, image: product.image ? `${SITE.url}${product.image}` : undefined, description: product.description, brand: { "@type": "Brand", name: brand.name }, manufacturer: { "@type": "Organization", name: brand.name, url: brand.website }, offers: { "@type": "Offer", url: `${SITE.url}/products/${product.slug}`, priceCurrency: "INR", itemCondition: "https://schema.org/NewCondition" } };
  if (product.availability) (schema.offers as Record<string, unknown>).availability = `https://schema.org/${product.availability}`;
  if (product.certifications?.countryOfOrigin) schema.countryOfOrigin = { "@type": "Country", name: product.certifications.countryOfOrigin };
  return schema;
}

export function generateArticleSchema(article: BlogArticle) {

import fs from "fs";
import path from "path";
import { SITE } from "../data/seo";

import { articles } from "../data/blog";

interface SitemapUrl { loc: string; lastmod?: string; changefreq?: string; priority?: number; }

const STATIC_ROUTES: SitemapUrl[] = [
  { loc: "/", changefreq: "weekly", priority: 1.0 },
  { loc: "/products", changefreq: "weekly", priority: 0.9 },
  { loc: "/blog", changefreq: "daily", priority: 0.9 },
  { loc: "/about", changefreq: "monthly", priority: 0.7 },
  { loc: "/contact", changefreq: "monthly", priority: 0.7 },
];

const FUTURE_ROUTES: SitemapUrl[] = [
  // { loc: "/case-studies", changefreq: "monthly", priority: 0.7 },
  // { loc: "/downloads", changefreq: "monthly", priority: 0.6 },
  // { loc: "/dealers", changefreq: "monthly", priority: 0.7 },
];

function generateSitemap(): string {
  const urls: SitemapUrl[] = [
    ...STATIC_ROUTES,
    ...products.map((p) => ({ loc: `/products/${p.slug}`, lastmod: p.updatedAt, changefreq: "weekly" as const, priority: 0.8 })),
    ...articles.filter((a) => a.status === "published").map((a) => ({ loc: `/blog/${a.slug}`, lastmod: a.date, changefreq: "monthly" as const, priority: 0.7 })),
  ];
  const urlEntries = urls.map((url) => `  <url>\n    <loc>${SITE.url}${url.loc}</loc>\n    ${url.lastmod ? `<lastmod>${url.lastmod}</lastmod>\n    ` : ""}${url.changefreq ? `<changefreq>${url.changefreq}</changefreq>\n    ` : ""}${url.priority ? `<priority>${url.priority.toFixed(1)}</priority>\n  ` : "  "}</url>`).join("\n");
  return `<?xml version="1.0" encoding="UTF-8"?>\n<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n${urlEntries}\n</urlset>`;
}

export function writeSitemap(outputDir: string = "public"): void {
  const sitemap = generateSitemap();
  const outputPath = path.resolve(outputDir, "sitemap.xml");
  fs.writeFileSync(outputPath, sitemap, "utf-8");
  console.log(`✓ Sitemap written to ${outputPath}`);
}

import { cn } from "@/lib/utils";

interface ImageProps extends React.ImgHTMLAttributes<HTMLImageElement> {
  src: string;
  alt: string;
  width: number;
  height: number;
  loading?: "eager" | "lazy";
  fetchpriority?: "high" | "low" | "auto";
  className?: string;
  caption?: string;
}

const GENERIC_ALTS = ["image", "photo", "picture", "img", ""];

export function Image({ src, alt, width, height, loading = "lazy", fetchpriority = "auto", className, caption, ...props }: ImageProps) {
  if (import.meta.env.DEV) {
    if (!alt || GENERIC_ALTS.includes(alt.toLowerCase().trim())) {
      throw new Error(`[Image SEO] Missing or generic alt text for: ${src}\nAlt must be descriptive. Current: "${alt}"`);
    }
    if (width < 10 || height < 10) {
      throw new Error(`[Image SEO] Invalid dimensions for: ${src} (${width}x${height})`);
    }
  }
  return (
    <figure className={cn("m-0", className)}>
      <img src={src} alt={alt} width={width} height={height} loading={loading} fetchPriority={fetchpriority} className="h-auto max-w-full" {...props} />
      {caption && <figcaption className="mt-2 text-center text-sm text-muted-foreground italic">{caption}</figcaption>}
    </figure>
  );
}
