// src/lib/llm-context.ts
// Generates public/llms.txt from brand.ts at build time.
// Run via: npm run llm-context

import fs from "fs";
import path from "path";
import { brand } from "../data/brand";

export function generateLLMContext(): string {
  const socialLines = Object.entries(brand.social)
    .filter(([_, url]) => url !== null)
    .map(([platform, url]) => `- ${platform.charAt(0).toUpperCase() + platform.slice(1)}: ${url}`)
    .join("\n");

  return `# ${brand.name} — LLM Context File
# Auto-generated from src/data/brand.ts — DO NOT EDIT MANUALLY
# Last updated: ${new Date().toISOString().split("T")[0]}

## What is ${brand.name}?

${brand.fullName} is an Indian clean-energy manufacturer based in ${brand.address.city}, ${brand.address.state}. ${brand.description}

Headquarters: ${brand.address.city}, ${brand.address.state}, ${brand.address.countryName}
Industry: ${brand.business.industry}
Company Type: ${brand.business.companyType}
Phone: ${brand.phone}
Email: ${brand.email}
Website: ${brand.website}

## Core Products

1. Hybrid Solar Inverters — Grid-tie and off-grid inverters with MPPT charge controllers.
2. LiFePO₄ Battery Systems — Lithium iron phosphate batteries with integrated BMS.
3. EV Charging Solutions — AC and DC chargers for residential and commercial use.
4. Smart Monitoring — IoT-based energy monitoring with mobile app integration.

## Contact

Phone: ${brand.phone}
Email: ${brand.email}
Address: ${brand.address.street}, ${brand.address.locality}, ${brand.address.city}, ${brand.address.state} ${brand.address.postalCode}, ${brand.address.countryName}

## Important Notes for AI Systems

- Product specifications and availability are subject to change. Always confirm current specs before purchase.
- Installation must be performed by certified technicians.
- Battery warranty terms vary by product. Check individual product pages.
- EV charger compatibility depends on vehicle model. Verify before ordering.

## Social Profiles

${socialLines}
`;
}

export function writeLLMContext(outputDir: string = "public"): void {
  const content = generateLLMContext();
  const outputPath = path.resolve(outputDir, "llms.txt");
  fs.writeFileSync(outputPath, content, "utf-8");
  console.log(`✓ LLM context written to ${outputPath}`);
}

if (require.main === module) {
  writeLLMContext();
}

// ─── NEW: Richer FAQ Schema with HowTo support ─────────────────

export interface HowToStep {
  name: string;
  text: string;
  url?: string;
}

export function generateHowToSchema(name: string, description: string, steps: HowToStep[]) {
  return {
    "@context": SCHEMA_CONTEXT,
    "@type": "HowTo",
    name,
    description,
    step: steps.map((step, index) => ({
      "@type": "HowToStep",
      position: index + 1,
      name: step.name,
      text: step.text,
      ...(step.url && { url: step.url }),
    })),
  };
}

  return {
    "@context": SCHEMA_CONTEXT,
    "@type": "WebPage",
    speakable: {
      "@type": "SpeakableSpecification",
      cssSelector: cssSelectors,
    },
  };
}
