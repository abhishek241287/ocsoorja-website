// src/lib/adapters/frontmatter.ts
// FrontMatter Adapter: parses YAML, knows NOTHING about Markdown.
// Single responsibility: extract key-value pairs from YAML frontmatter.

export interface FrontMatterResult {
  [key: string]: unknown;
}

export function parseFrontMatter(raw: string): { frontmatter: FrontMatterResult; content: string } {
  const match = raw.match(/^---\n([\s\S]*?)\n---\n([\s\S]*)$/);
  if (!match) return { frontmatter: {}, content: raw };

  const frontmatter: FrontMatterResult = {};
  const lines = match[1].split("\n");
  let currentKey = "";
  let currentValue: unknown = "";
  let inArray = false;

  for (const line of lines) {
    const trimmed = line.trim();
    const keyMatch = line.match(/^(\w+):\s*(.*)$/);

    if (keyMatch && !line.startsWith(" ")) {
      if (currentKey) frontmatter[currentKey] = currentValue;
      currentKey = keyMatch[1];
      const value = keyMatch[2].trim();

      if (value === "") {
        currentValue = "";
        inArray = false;
      } else if (value.startsWith("[") && value.endsWith("]")) {
        currentValue = value.slice(1, -1).split(",").map((v) => v.trim().replace(/^["']|["']$/g, ""));
        inArray = false;
      } else if (value === "true") {
        currentValue = true;
        inArray = false;
      } else if (value === "false") {
        currentValue = false;
        inArray = false;
      } else if (!isNaN(Number(value)) && value !== "") {
        currentValue = Number(value);
        inArray = false;
      } else {
        currentValue = value.replace(/^["']|["']$/g, "");
        inArray = false;
      }
      continue;
    }

    const nestedMatch = line.match(/^  (\w+):\s*(.*)$/);
    if (nestedMatch && currentKey && (typeof currentValue === "string" && currentValue === "" || (typeof currentValue === "object" && currentValue !== null && !Array.isArray(currentValue)))) {
      const nestedKey = nestedMatch[1];
      const nestedValue = nestedMatch[2].trim().replace(/^["']|["']$/g, "");
      if (typeof currentValue === "string") {
        currentValue = { [nestedKey]: nestedValue };
      } else {
        (currentValue as Record<string, unknown>)[nestedKey] = nestedValue;
      }
      continue;
    }

    if (trimmed.startsWith("- ") && currentKey) {
      const item = trimmed.replace(/^- /, "").replace(/^["']|["']$/g, "");
      if (!inArray) {
        currentValue = [item];
        inArray = true;
      } else if (Array.isArray(currentValue)) {
        currentValue.push(item);
      }
    }
  }

  if (currentKey) frontmatter[currentKey] = currentValue;

  return { frontmatter, content: match[2].trim() };
}

// src/lib/adapters/markdown.ts
// Markdown Adapter: parses Markdown into AST.
// Single responsibility: convert Markdown text to structured AST nodes.
// Knows NOTHING about YAML frontmatter.

import type { MarkdownNode, TableOfContentsItem } from "@/types/blog";

export interface MarkdownParseResult {
  ast: MarkdownNode[];
  toc: TableOfContentsItem[];
  wordCount: number;
  readingTime: number;
}

export function parseMarkdown(content: string): MarkdownParseResult {
  const ast: MarkdownNode[] = [];
  const toc: TableOfContentsItem[] = [];
  let wordCount = 0;

  const blocks = content.split(/\n\n+/);

  for (const block of blocks) {
    const trimmed = block.trim();
    if (!trimmed) continue;

    // Heading
    const headingMatch = trimmed.match(/^(#{2,3})\s+(.+)$/);
    if (headingMatch) {
      const level = headingMatch[1].length as 2 | 3;
      const text = headingMatch[2].trim();
      const id = text.toLowerCase().replace(/[^\w]+/g, "-").replace(/^-|-$/g, "");
      toc.push({ id, text, level });
      ast.push({
        type: "heading",
        tag: `h${level}`,
        props: { id },
        children: parseInline(text),
      });
      wordCount += text.split(/\s+/).length;
      continue;
    }

    // Blockquote
    if (trimmed.startsWith(">")) {
      const text = trimmed.replace(/^>\s*/, "").trim();
      ast.push({
        type: "blockquote",
        children: parseInline(text),
      });
      wordCount += text.split(/\s+/).length;
      continue;
    }

    // Code block
    const codeMatch = trimmed.match(/^```(\w+)?\n([\s\S]*?)```$/);
    if (codeMatch) {
      ast.push({
        type: "code",
        props: { lang: codeMatch[1] || "" },
        content: codeMatch[2].trim(),
      });
      continue;
    }

    // Unordered list
    if (trimmed.split("\n").every((line) => line.trim().startsWith("- "))) {
      const items = trimmed.split("\n").map((line) => {
        const text = line.trim().replace(/^- /, "");
        return {
          type: "listItem" as const,
          children: parseInline(text),
        };
      });
      ast.push({
        type: "list",
        props: { ordered: false },
        children: items,
      });
      wordCount += items.reduce((sum, item) => sum + (typeof item.children[0] === "string" ? item.children[0].split(/\s+/).length : 0), 0);
      continue;
    }

    // Ordered list
    if (trimmed.split("\n").every((line) => /^\d+\.\s+/.test(line.trim()))) {
      const items = trimmed.split("\n").map((line) => {
        const text = line.trim().replace(/^\d+\.\s+/, "");
        return {
          type: "listItem" as const,
          children: parseInline(text),
        };
      });
      ast.push({
        type: "list",
        props: { ordered: true },
        children: items,
      });
      wordCount += items.reduce((sum, item) => sum + (typeof item.children[0] === "string" ? item.children[0].split(/\s+/).length : 0), 0);
      continue;
    }

    // Paragraph
    ast.push({
      type: "paragraph",
      children: parseInline(trimmed),
    });
    wordCount += trimmed.split(/\s+/).length;
  }

  const readingTime = Math.max(1, Math.ceil(wordCount / 200));

  return { ast, toc, wordCount, readingTime };
}

function parseInline(text: string): (MarkdownNode | string)[] {
  const result: (MarkdownNode | string)[] = [];
  let remaining = text;

  // Image
  const imageRegex = /!\[([^\]]*)\]\(([^)]+)\)/g;
  let match;
  while ((match = imageRegex.exec(text)) !== null) {
    const before = text.slice(remaining.indexOf(text) === 0 ? 0 : text.indexOf(remaining), match.index);
    if (before) result.push(...parseTextWithFormatting(before));
    result.push({
      type: "image",
      props: { src: match[2], alt: match[1] },
    });
    remaining = text.slice(match.index + match[0].length);
  }
  if (remaining) result.push(...parseTextWithFormatting(remaining));

  return result.length > 0 ? result : parseTextWithFormatting(text);
}

function parseTextWithFormatting(text: string): (MarkdownNode | string)[] {
  const result: (MarkdownNode | string)[] = [];
  let remaining = text;

  // Link
  const linkRegex = /\[([^\]]+)\]\(([^)]+)\)/g;
  let match;
  while ((match = linkRegex.exec(text)) !== null) {
    const before = text.slice(0, match.index);
    if (before) result.push(before);
    result.push({
      type: "link",
      props: { href: match[2] },
      children: [match[1]],
    });
    remaining = text.slice(match.index + match[0].length);
  }
  if (remaining && remaining !== text) {
    result.push(remaining);
    return result;
  }

  // Bold + Italic
  const boldItalicRegex = /\*\*\*(.+?)\*\*\*/g;
  while ((match = boldItalicRegex.exec(text)) !== null) {
    const before = text.slice(0, match.index);
    if (before) result.push(before);
    result.push({
      type: "strong",
      children: [{ type: "em", children: [match[1]] }],
    });
    remaining = text.slice(match.index + match[0].length);
  }
  if (remaining && remaining !== text) return result;

  // Bold
  const boldRegex = /\*\*(.+?)\*\*/g;
  while ((match = boldRegex.exec(text)) !== null) {
    const before = text.slice(0, match.index);
    if (before) result.push(before);
    result.push({
      type: "strong",
      children: [match[1]],
    });
    remaining = text.slice(match.index + match[0].length);
  }
  if (remaining && remaining !== text) return result;

  // Italic
  const italicRegex = /\*(.+?)\*/g;
  while ((match = italicRegex.exec(text)) !== null) {
    const before = text.slice(0, match.index);
    if (before) result.push(before);
    result.push({
      type: "em",
      children: [match[1]],
    });
    remaining = text.slice(match.index + match[0].length);
  }
  if (remaining && remaining !== text) return result;

  // Inline code
  const codeRegex = /`([^`]+)`/g;
  while ((match = codeRegex.exec(text)) !== null) {
    const before = text.slice(0, match.index);
    if (before) result.push(before);
    result.push({
      type: "inlineCode",
      content: match[1],
    });
    remaining = text.slice(match.index + match[0].length);
  }
  if (remaining && remaining !== text) return result;

  return [text];
}

// src/lib/adapters/renderer.ts
// Renderer: converts AST to React elements.
// Single responsibility: AST → React. No Markdown knowledge. No YAML knowledge.

import React from "react";
import type { MarkdownNode } from "@/types/blog";

export function renderAST(ast: MarkdownNode[]): React.ReactNode[] {
  return ast.map((node, index) => renderNode(node, index));
}

function renderNode(node: MarkdownNode, key: number): React.ReactNode {
  if (typeof node === "string") return node;

  const children = node.children ? renderAST(node.children as MarkdownNode[]) : node.content;

  switch (node.type) {
    case "heading":
      return React.createElement(node.tag || "h2", { key, ...node.props }, children);
    case "paragraph":
      return React.createElement("p", { key, className: "my-4 leading-relaxed" }, children);
    case "blockquote":
      return React.createElement("blockquote", { key, className: "border-l-4 border-primary pl-4 italic text-muted-foreground my-4" }, children);
    case "code":
      return React.createElement("pre", { key, className: "bg-muted p-4 rounded-lg overflow-x-auto my-4" },
        React.createElement("code", { className: "text-sm" }, node.content)
      );
    case "list":
      const listTag = node.props?.ordered ? "ol" : "ul";
      const listClass = node.props?.ordered ? "list-decimal" : "list-disc";
      return React.createElement(listTag, { key, className: `my-4 ${listClass} pl-6 space-y-1` }, children);
    case "listItem":
      return React.createElement("li", { key }, children);
    case "image":
      return React.createElement("figure", { key, className: "my-6" },
        React.createElement("img", {
          src: node.props?.src as string,
          alt: node.props?.alt as string,
          className: "rounded-lg w-full",
          loading: "lazy",
        }),
        node.props?.alt && React.createElement("figcaption", { className: "mt-2 text-center text-sm text-muted-foreground" }, node.props?.alt)
      );
    case "link":
      return React.createElement("a", { key, href: node.props?.href as string, className: "text-primary hover:underline" }, children);
    case "strong":
      return React.createElement("strong", { key }, children);
    case "em":
      return React.createElement("em", { key }, children);
    case "inlineCode":
      return React.createElement("code", { key, className: "bg-muted px-1 py-0.5 rounded text-sm" }, node.content);
    case "text":
      return node.content || "";
    default:
      return React.createElement("span", { key }, children);
  }
}

// Server-side / build-time HTML renderer (no React dependency)
export function renderASTToHTML(ast: MarkdownNode[]): string {
  return ast.map((node) => nodeToHTML(node)).join("");
}

function nodeToHTML(node: MarkdownNode): string {
  if (typeof node === "string") return escapeHtml(node);

  const children = node.children 
    ? (node.children as MarkdownNode[]).map(nodeToHTML).join("")
    : node.content || "";

  switch (node.type) {
    case "heading":
      return `<${node.tag} ${propsToAttrs(node.props)}>${children}</${node.tag}>`;
    case "paragraph":
      return `<p class="my-4 leading-relaxed">${children}</p>`;
    case "blockquote":
      return `<blockquote class="border-l-4 border-primary pl-4 italic text-muted-foreground my-4">${children}</blockquote>`;
    case "code":
      return `<pre class="bg-muted p-4 rounded-lg overflow-x-auto my-4"><code class="text-sm">${escapeHtml(node.content || "")}</code></pre>`;
    case "list":
      const tag = node.props?.ordered ? "ol" : "ul";
      const cls = node.props?.ordered ? "list-decimal" : "list-disc";
      return `<${tag} class="my-4 ${cls} pl-6 space-y-1">${children}</${tag}>`;
    case "listItem":
      return `<li>${children}</li>`;
    case "image":
      return `<figure class="my-6"><img src="${node.props?.src}" alt="${escapeHtml(node.props?.alt as string)}" class="rounded-lg w-full" loading="lazy" /><figcaption class="mt-2 text-center text-sm text-muted-foreground">${escapeHtml(node.props?.alt as string)}</figcaption></figure>`;
    case "link":
      return `<a href="${node.props?.href}" class="text-primary hover:underline">${children}</a>`;
    case "strong":
      return `<strong>${children}</strong>`;
    case "em":
      return `<em>${children}</em>`;
    case "inlineCode":
      return `<code class="bg-muted px-1 py-0.5 rounded text-sm">${escapeHtml(node.content || "")}</code>`;
    case "text":
      return escapeHtml(node.content || "");
    default:
      return `<span>${children}</span>`;
  }
}

function propsToAttrs(props?: Record<string, unknown>): string {
  if (!props) return "";
  return Object.entries(props)
    .map(([k, v]) => `${k}="${escapeHtml(String(v))}"`)
    .join(" ");
}

function escapeHtml(text: string): string {
  return text
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#039;");
}

// src/lib/content-engine.ts
// Content Engine: orchestrates FrontMatter + Markdown + Renderer.
// Exposes parse(), render(), validate(). Does NOT know YAML or Markdown details.

import { parseFrontMatter } from "./adapters/frontmatter";
import { parseMarkdown } from "./adapters/markdown";
import { renderASTToHTML } from "./adapters/renderer";
import type { BlogArticle, ParsedArticle, MarkdownNode, TableOfContentsItem, ArticleValidationResult } from "@/types/blog";
import { brand } from "@/data/brand";

export interface ContentEngine {
  parse(raw: string, fileId: string): ParsedArticle;
  render(ast: MarkdownNode[]): { html: string; toc: TableOfContentsItem[] };
  validate(article: Partial<BlogArticle>): ArticleValidationResult;
}

class DefaultContentEngine implements ContentEngine {
  parse(raw: string, fileId: string): ParsedArticle {
    // Step 1: FrontMatter Adapter extracts YAML
    const { frontmatter, content } = parseFrontMatter(raw);

    // Step 2: Markdown Adapter parses content to AST
    const { ast, toc, wordCount, readingTime } = parseMarkdown(content);

    // Step 3: Build BlogArticle from frontmatter + AST stats
    const article = this.buildArticle(frontmatter, fileId, content, wordCount, readingTime);

    return { article, ast, toc };
  }

  render(ast: MarkdownNode[]): { html: string; toc: TableOfContentsItem[] } {
    const html = renderASTToHTML(ast);
    // TOC is extracted during parse, not render
    return { html, toc: [] };
  }

  validate(article: Partial<BlogArticle>): ArticleValidationResult {
    const errors: string[] = [];
    const warnings: string[] = [];

    if (!article.id || article.id.trim().length < 3) {
      errors.push("ID is required and must be at least 3 characters (use meaningful IDs like 'solar-home-savings').");
    }
    if (!article.title || article.title.trim().length < 5) {
      errors.push("Title is required and must be at least 5 characters.");
    }
    if (!article.content || article.content.trim().length < 100) {
      errors.push("Content is required and must be at least 100 characters.");
    }
    if (!article.slug || article.slug.trim().length === 0) {
      errors.push("Slug is required.");

// src/lib/content-cache.ts
// File-based incremental cache with content hashing.
// Only rebuilds articles that changed.

import fs from "fs";
import path from "path";
import type { ParsedArticle } from "@/types/blog";

const CACHE_DIR = ".content-cache";

interface CacheManifest {
  version: string;
  generatedAt: string;
  articles: Record<string, { hash: string; filePath: string }>;
}

function ensureCacheDir(): void {
  if (!fs.existsSync(CACHE_DIR)) {
    fs.mkdirSync(CACHE_DIR, { recursive: true });
  }
}

function getManifestPath(): string {
  return path.join(CACHE_DIR, "manifest.json");
}

function getArticleCachePath(id: string): string {
  return path.join(CACHE_DIR, "articles", `${id}.json`);
}

function getHashesPath(): string {
  return path.join(CACHE_DIR, "hashes.json");
}

function loadManifest(): CacheManifest | null {
  const manifestPath = getManifestPath();
  if (!fs.existsSync(manifestPath)) return null;
  return JSON.parse(fs.readFileSync(manifestPath, "utf-8"));
}

function saveManifest(manifest: CacheManifest): void {
  fs.writeFileSync(getManifestPath(), JSON.stringify(manifest, null, 2), "utf-8");
}

function computeHash(content: string): string {
  let hash = 0;
  for (let i = 0; i < content.length; i++) {
    const char = content.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  return hash.toString(16);
}

export function loadCachedArticle(id: string): ParsedArticle | null {
  const cachePath = getArticleCachePath(id);
  if (!fs.existsSync(cachePath)) return null;
  return JSON.parse(fs.readFileSync(cachePath, "utf-8"));
}

export function saveCachedArticle(id: string, article: ParsedArticle): void {
  ensureCacheDir();
  const articlesDir = path.join(CACHE_DIR, "articles");
  if (!fs.existsSync(articlesDir)) {
    fs.mkdirSync(articlesDir, { recursive: true });
  }
  fs.writeFileSync(getArticleCachePath(id), JSON.stringify(article, null, 2), "utf-8");
}

export function getCachedArticles(): ParsedArticle[] {
  const articlesDir = path.join(CACHE_DIR, "articles");
  if (!fs.existsSync(articlesDir)) return [];
  
  return fs.readdirSync(articlesDir)
    .filter((f) => f.endsWith(".json"))
    .map((f) => {
      const content = fs.readFileSync(path.join(articlesDir, f), "utf-8");
      return JSON.parse(content) as ParsedArticle;
    });
}

export function getChangedArticles(rawArticles: Record<string, string>): string[] {
  const manifest = loadManifest();
  const hashesPath = getHashesPath();
  const currentHashes: Record<string, string> = {};
  
  for (const [id, content] of Object.entries(rawArticles)) {
    currentHashes[id] = computeHash(content);
  }

  // Save current hashes
  fs.writeFileSync(hashesPath, JSON.stringify(currentHashes, null, 2), "utf-8");

  if (!manifest) return Object.keys(rawArticles);

  const changed: string[] = [];
  for (const [id, hash] of Object.entries(currentHashes)) {
    const cached = manifest.articles[id];
    if (!cached || cached.hash !== hash) {
      changed.push(id);
    }
  }

  return changed;
}

export function updateManifest(articles: Record<string, ParsedArticle>): void {
  const manifest: CacheManifest = {
    version: "1.0",
    generatedAt: new Date().toISOString(),
    articles: {},
  };

  for (const [id, parsed] of Object.entries(articles)) {
    manifest.articles[id] = {
      hash: computeHash(parsed.article.content),
      filePath: getArticleCachePath(id),
    };
  }

  saveManifest(manifest);
}

export function clearCache(): void {
  if (fs.existsSync(CACHE_DIR)) {
    fs.rmSync(CACHE_DIR, { recursive: true });
  }
}

  });

// ─── Load All Articles (with incremental cache) ────────────────

export function loadAllArticles(): ParsedArticle[] {
  const rawArticles = loadRawArticles();
  const changedIds = getChangedArticles(rawArticles);
  const engine = getContentEngine();

  const parsedArticles: Record<string, ParsedArticle> = {};

  // Load unchanged from cache
  for (const [id, raw] of Object.entries(rawArticles)) {
    if (!changedIds.includes(id)) {
      const cached = loadCachedArticle(id);
      if (cached) {
        parsedArticles[id] = cached;
        continue;
      }
    }

    // Parse changed/new articles
    try {
      const parsed = engine.parse(raw, id);
      saveCachedArticle(id, parsed);
      parsedArticles[id] = parsed;
    } catch (error) {
      console.error(`Failed to parse article ${id}:`, error);
    }
  }

  // Update manifest
  updateManifest(parsedArticles);

  // Sort by publish date descending
  const result = Object.values(parsedArticles);
  result.sort((a, b) => new Date(b.article.publishDate).getTime() - new Date(a.article.publishDate).getTime());

  return result;
}

        score += sharedKeywords * 2;

        score += sharedTopics * 3;
        const sharedEntities = (candidate.article.entityMetadata?.entities || []).filter((e) => (article.entityMetadata?.entities || []).includes(e)).length;
        score += sharedEntities * 2;
        return { id: candidate.article.id, score };
      });

      scored.sort((a, b) => b.score - a.score);
      const top3 = scored.slice(0, 3).map((s) => s.id);

function escapeXml(str: string): string {
  return str
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&apos;");
}

function extractEntitiesFromAST(ast: any[]): string[] {
  const entities: string[] = [];
  for (const node of ast) {
    if (typeof node === "string") {
      // Simple entity extraction: look for capitalized terms
      const matches = node.match(/\b[A-Z][a-zA-Z]*(?:\s+[A-Z][a-zA-Z]*)+\b/g);
      if (matches) entities.push(...matches);
    }
    if (node.children) {
      entities.push(...extractEntitiesFromAST(node.children));
    }
  }
  return [...new Set(entities)];
}
