Final Specification: OCS OORJA Projects & Case Studies
Core Philosophy
This is not a photo gallery. It is the portfolio of a serious engineering company. Every project page is a Case Study that demonstrates expertise, builds trust, and drives commercial intent.
Data Model (TypeScript)
TypeScript
// src/types/projects.ts

export type ProjectStatus = "completed" | "commissioning" | "in-progress" | "upcoming";

export type IndustrySegment = "Residential" | "Commercial" | "Industrial" | "Agriculture" | "EV";

export type ProjectTechnology = 
  | "LiFePO4" 
  | "Hybrid Inverter" 
  | "BMS" 
  | "MPPT" 
  | "Solar" 
  | "EV Charger" 
  | "Smart Monitoring";

export interface ProjectTimeline {
  survey?: string;      // ISO date
  engineering?: string;
  installation?: string;
  commissioning?: string;
  handover?: string;
}

export interface BeforeAfterImage {
  before: {
    src: string;
    alt: string;
    caption?: string;
  };
  after: {
    src: string;
    alt: string;
    caption?: string;
  };
}

export interface ROIData {
  estimatedMonthlySavings?: number;    // INR
  annualSavings?: number;              // INR
  backupDuration?: number;             // hours
  estimatedPayback?: number;           // years
  co2Reduction?: number;               // kg/year
  energyGenerated?: number;            // kWh/year
}

export interface ProjectCaseStudy {
  // ─── Core Identity ───
  id: string;              // Meaningful: "lucknow-residential-5kw"
  slug: string;
  title: string;
  subtitle?: string;       // "5kW Hybrid Solar + 10kWh Battery for 3BHK Home"
  
  // ─── Status & Classification ───
  status: ProjectStatus;
  segment: IndustrySegment;
  location: {
    city: string;
    state: string;
    country: string;      // "IN"
  };
  
  // ─── Dates ───
  completedDate?: string;  // ISO 8601
  publishDate: string;
  updatedAt: string;
  
  // ─── Client Info ───
  clientName?: string;     // "Confidential" or actual name with permission
  clientType: "Homeowner" | "Small Business" | "Enterprise" | "Government" | "Educational";
  
  // ─── Technical Specs ───
  systemSize: {
    solarKw: number;
    batteryKwh: number;
    inverterKw: number;
  };
  technologies: ProjectTechnology[];
  productsUsed: string[];  // Product IDs from product catalog
  
  // ─── ROI & Performance ───
  roi?: ROIData;
  
  // ─── Challenge & Solution ───
  challenge: string;       // What problem did the client face?
  solution: string;        // How did OCS OORJA solve it?
  results: string;         // What measurable outcomes were achieved?
  
  // ─── Visual Content ───
  heroImage: {
    src: string;
    alt: string;
    width: number;
    height: number;
  };
  galleryImages: {
    src: string;
    alt: string;
    caption?: string;
    category: "installation" | "commissioning" | "monitoring" | "team" | "site";
  }[];
  beforeAfter?: BeforeAfterImage[];
  
  // ─── Timeline ───
  timeline?: ProjectTimeline;
  
  // ─── Downloads & Links ───
  brochureUrl?: string;
  caseStudyPdf?: string;
  testimonialId?: string;  // Link to future testimonial
  
  // ─── SEO & AI ───
  seoTitle?: string;
  metaDescription?: string;
  keywords?: string[];
  summary: string;         // For AI extraction
  directAnswer: string;    // "What did this project achieve?"
  
  // ─── Related Content ───
  relatedArticleIds?: string[];
  relatedProductIds?: string[];
  
  // ─── Team & Attribution ───
  projectEngineer?: string;
  installationTeam?: string[];
  
  // ─── Schema.org ───
  schemaType: "Project" | "CaseStudy";
}

Lighthouse Best Practices (Not Fixed Targets)
Instead of fixed scores, follow these practices and measure results:
Images: WebP/AVIF, lazy loading, proper dimensions, descriptive alt text
Fonts: Preload critical fonts, use font-display: swap
CSS/JS: Code splitting, tree shaking, minimal critical CSS
Caching: Static asset caching, ISR for project pages
Accessibility: Semantic HTML, ARIA labels, keyboard navigation, color contrast
Core Web Vitals: Measure LCP, FID, CLS in production
Mobile: Responsive design, touch targets, readable fonts
Measure with: Chrome DevTools Lighthouse, PageSpeed Insights, Web Vitals RUM.
